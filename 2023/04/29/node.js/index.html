Node.JS 是Js的后端环境，无法调用DOM和BOM等浏览器内置API

  4. 读写和操作数据库、创建使用的命令行开发工具辅助前端开发、etc等

四、fs文件系统模块

1.概念

fs模块 是官方提供的、用来操作文件的模块。他提供了一系列的方法和属性，用来满足用户对文件的需求。

2.使用

导入fs模块

const fs = require('fs')

2.常用方法

  • fs.readFile( )

    fs.readFile(path[,options],callback)
//path: 文件路径 必选
//options ： 编码格式 可选
//callback ： 通过回调函数拿到读取结果 必选
    const fs = require('fs')
// err 错误对象
// dataStr 成功的结果
//读取成功，err 值为 null
//读取失败 ， err 值为 错误对象 ，dataStr 为 undefined
fs.readFile('F:/Code_Vue/code/2.txt','utf8',function(err,dataStr){

    if(err){
        return console.log("读取失败"+err.message);
    }
    console.log("读取成功，文件 = "+dataStr);
})

  • fs.writeFile( )

    fs.writeFile(file,data[,options],callback)
//file: 文件路径的字符串 必选
//data : 写入内容 必选
//options ：  编码格式 可选
//callback ： 通过回调函数拿到读取结果 必选
    const fs = require("fs");

fs.writeFile('F:/Code_Vue/code/3.txt','abcd123',function(err,dataStr){
    if(err){
        return console.log("写入失败"+err.message);
    }
    console.log("写入成功，文件 = "+dataStr);
})

案例：读取文件，输出文件

旧文件：

小红=99 小白=100 小李=29 小绿=69

新文件：

小红: 99
小白: 100
小李: 29
小绿: 69

代码：

const fs = require("fs");

fs.readFile('F:/Code_Vue/code/成绩.txt','utf8',function(err,dataStr){
    if(err){
        return console.log("读取失败"+err);
    }
    // console.log("读取成功 = "+dataStr);
    //	根据分隔符，把字符串转换为数组
    const arr = dataStr.split(' ');
    //使用数组接收
    const arrNew = []
    //使用forEach操作每个元素，push更改，replace替换
    arr.forEach(item =>{
        arrNew.push(item.replace('=',': '));
    })
    //join合并字符串，\r\n回车换行
    const newStr = arrNew.join('\r\n');
    console.log(newStr);
    fs.writeFile('F:/Code_Vue/code/成绩-ok.txt',newStr,function(err,dataStr){
        if(err){
            return console.log("写入失败"+err.message);
        }
        console.log("写入成功，文件 = "+dataStr);
    })
})

3.路径问题（路径拼接）

const fs = require('fs')
//相对路径，与运行的cmd窗口的文件目录有关（node 命令所在的文件目录），不一致报错
//./自动拼接cmd窗口的文件目录
//使用绝对路径,但是移植性差，建议使用__dirname
fs.readFile('./成绩.txt','utf8',function(err,dataStr){
    console.log(err);
    console.log(dataStr);
})

使用绝对路径,但是移植性差，可使用__dirname，建议使用path模块

const fs = require('fs');

fs.readFile(__dirname+'/成绩.txt','utf8',function(err,dataStr){
    console.log(err);
    console.log(dataStr);
})

五、path路径模块

1.概念

path模块由官方提供，用来处理路径的模块，提供了一系列的的方法和属性，满足用户对路径的处理需求。

2.使用

导入path模块

const path = require('path')

3.常用方法

  • path.join() 拼接路径

    const path = require('path')

//注意 ../ 会抵消前一个层径
const pathStr = path.join('/a','/b/c','../','./d','/e')
//\a\b\d\e
console.log(pathStr);

!!! 注意： 在进行路径拼接时，使用 path.join（），不要使用 + 进行字符串拼接

const path = require('path')
const fs = require('fs')

fs.readFile(path.join(__dirname,'/成绩.txt',),'utf8',function(err,dataStr){
    if(err){}
    console.log(dataStr);
})

  • path.basename() 获取路径中的文件名

    path.basename(path[,ext]);
//path 路径字符串 必选
//ext 文件扩展名 可选
const path = require('path')

var fpath = 'a/b/c/index.html'
console.log(fpath)
//a/b/c/index.html
var pp = path.basename(fpath,'.html')
console.log(pp)
//index

  • path.extname() 获取路径中的文件扩展名

    path.extname(path)
const path = require('path')


var fpath = 'a/b/c/index.html'
console.log(fpath)
//a/b/c/index.html
var pp = path.extname(fpath)
console.log(pp)
//.html

七、Node.js 模块化

1.作用域

和函数作用域类似，在自定义模块中定义的变量、方法等成员，只能在当前模块访问

2.如何向外共享成员

1.module对象

Module {
  id: '.',
  path: 'F:\\Code_Vue\\code',
  exports: {},
  filename: 'F:\\Code_Vue\\code\\t1.js',
  loaded: false,
  children: [],
  paths: [
    'F:\\Code_Vue\\code\\node_modules',
    'F:\\Code_Vue\\node_modules',
    'F:\\node_modules'
  ]
}

2.module.exports 对象

自定义模块中，使用module.exports对象，将模块中的成员共享出去，供外界使用

外界使用require（） 导入自定义模块时，得到的是module.exports所指向的对象

const age = 20

//为module.exports对象挂载username属性
module.exports.username = 'zs'
//为module.exports对象挂载 sayHello 方法
module.exports.sayHello = function(){
    console.log('Hello')
}
module.exports.age = age
//优先显示
module.exports = {
    nickname: '123',
    sayHi(){
        console.log('111');
    }
}

3.exports对象

exports与 module.exports指向同一对象，但是最终结果以 module.exports指向的对象为准

//输出
module.exports = {
    nickname: '123',
    sayHi(){
        console.log('111');
    }
}

exports = {
    nickname: '456',
    sayHi(){
        console.log('111');
    }
}

console.log(exports === module.exports) //false

3.模块化规范

  • 模块内部，module变量代表当前模块
  • module变量是一个对象，module.exports 是对外的接口
  • 加载某个模块，其实是加载该模块的module.exports 属性，require（）用于加载模块

4.npm包的使用

1.不使用npm包，格式化时间

//clock.js
function dataFormat(deStr){
    const dt = new Date(deStr);

    const y = dt.getFullYear();
    const m = padZero(dt.getMonth() +1);
    const d = padZero(dt.getDate());

    const hh = padZero(dt.getHours());
    const mm = padZero(dt.getHours());
    const ss = padZero(dt.getSeconds());
    return `${y}-${m}-${d}-${hh}-${mm}-${ss}`
}
//补零
function padZero(n){
    return n > 9 ?  n : '0'+n;
}
module.exports  = {
    dataFormat
}

九、express

1.什么是Express？

基于Node.js平台创建的Web开发框架。

2.基本使用

  • 安装

    npm i express

  • 创建基本的Web服务器

    const express = require('express')

app.listen(8080,()=>{
    console.log("App is Running at http:127.0.0.1:8080");
})

  • 监听Get和Post请求

6.中间件

1.概念：

业务流程的中间处理环节。

本质：function()函数,里面有next属性

app.get('/',function(req,res,next)=>{
         next():
})

2.next函数的作用

实现多个中间件连续调用的关键，表示把流转关系转交给下一个中间件或路由。

3.定义中间件函数

1.全局中间件：

每次请求都调用

//定义中间件函数
const mw = function(req,res,next){
    console.log('中间件函数')
    next();
}
//注册为全局中间件
app.use(mw)
//=======================

//定义并注册为全局中间件
app.use(function(req,res,next){
    console.log('中间件函数')
    next();
})

案例：

const express = require('express')

const app = express();

//定义中间件函数
const mw = function(req,res,next){
    console.log('中间件函数')
    next();
}
//注册为全局中间件
app.use(mw)

app.get('/',(req,res)=>{
    res.send('Home page')
})
app.get('/user',(req,res)=>{
    res.send('User Page.')
})
app.listen(8080,()=>{
    console.log('App is Running At 127.0.0.1:8080')
})

2.局部中间件：

//定义中间件函数
const mw = function(req,res,next){
    console.log('中间件函数')
    next();
}
//可写多个
app.get('/',mw,(req,res)=>{
    res.send('Home page')
})

4.中间件的作用和特点：

特点：多个中间件共享同一份req和res。

作用：类似于（代码块）

5.使用注意事项

  • 在路由之前注册中间件
  • 请求可以通过多个中间件处理
  • 执行完中间件的业务代码，不要忘记调用next()函数
  • 防止代码逻辑混乱，调用next() 之后不要写任何代码
  • 连续调用多个中间件，req，res共享的

6.中间件分类

  • 应用级别的中间件

    app.use(),app.get(),app.post()

  • 路由级别的中间件
